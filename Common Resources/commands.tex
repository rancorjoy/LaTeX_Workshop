%Chapter spacing command:
\newcommand{\hsp}{\hspace{20pt}}

\titlespacing*{\chapter}{0pt}{0pt}{40pt}

\newcommand{\DrawPowerRails}[4] %Draws Breadboard Power Rails - must be in CircuitTikz!
{
\draw [red, line width = 0.75 mm](#1, #4) -- (#2, #4) node [red, right]{$+V_{CC}$};
\draw [blue, line width = 0.75 mm](#1, #3) -- (#2, #3) node [blue, right]{$Gnd$};
}

\newcommand{\TwoColumns}[2] %Shortcut for multicols setup
{
\begin{multicols}{2}
#1
\columnbreak
#2
\end{multicols}
}

\newcommand{\ab} %Creates (a) (b) for two column figures
{
\begin{multicols}{2}
\center{(a)}\columnbreak\center{(b)}
\end{multicols}
}

\newcommand{\nopad}{\vspace{-0.85\baselineskip}}

%Command to render Cyrillic Character Sha (for Dirac Delta Comb)
\newcommand{\Sha}{%
  \tikz[baseline=(I.base)]{
    \node[inner sep=0pt] (I) {\textls[0]{III}};
    \draw[line width=0.2pt] (I.south west) -- (I.south east);
  }%
}

%In Line Listing Environment
\newcommand{\inlineListing}[1]{\colorbox{gray!25}{\lstinline!#1!}}

\newcommand{\inlinemathListing}[1]{\colorbox{gray!25}{\lstinline[mathescape,keepspaces]!#1!}}



\newcommand{\SetCodeLang}[1]
{
\lstset{
    language=#1,
    backgroundcolor=\color{backcolour},   
    commentstyle=\color{blue},
    keywordstyle=\color{magenta},
    numberstyle=\tiny\color{codegray},
    stringstyle=\color{codepurple},
    basicstyle=\ttfamily\footnotesize,
    breakatwhitespace=false,         
    breaklines=true,                 
    captionpos=b,
    columns=fullflexible,                    
    keepspaces=true,                 
    numbers=left,                    
    numbersep=5pt,                  
    showspaces=false,                
    showstringspaces=false,
    showtabs=false,                  
    tabsize=2
}}

%Custom Code Block Environment
\newtcblisting{codeBox}[1][]{%
  colback=gray!5,
  colframe=gray!15,
  boxsep=5pt,
  left=5pt,
  right=5pt,
  top=5pt,
  bottom=5pt,
  listing only,
  listing engine=minted,
  minted language=python,
  minted options={%
    fontsize=\footnotesize,
    baselinestretch=1.2,
    breaklines
  },
  before={\vspace{0.5\baselineskip}\noindent},
  after={\vspace{0.5\baselineskip}},
  #1
}
